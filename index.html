<script>


function $(id){ return document.getElementById(id); }


let configuracions=[null,null,null];

let ranking=[0,0,0,0];

let rondaGlobal=0;

let rondaCalculada=false;


function guardar(){


let f=Math.floor(rondaGlobal/8);


configuracions[f]={

t1V:Number($("t1V").value),

t1H:Number($("t1H").value),

t2V:Number($("t2V").value),

t2H:Number($("t2H").value)

};


alert("Configuració guardada");

}


function getJ(){


let j1=document.querySelector('input[name="j1"]:checked');

let j2=document.querySelector('input[name="j2"]:checked');


if(!j1 || !j2){

alert("Falta indicar qui juga");

return null;

}


return [

[j1.value==="V"?1:0, j1.value==="H"?1:0],

[j2.value==="V"?1:0, j2.value==="H"?1:0]

];

}


function rival(x,config){

if(x===config.t1V) return config.t1H;

if(x===config.t1H) return config.t1V;

if(x===config.t2V) return config.t2H;

return config.t2V;

}


function calcular(){


if(rondaCalculada){

alert("Ja calculada");

return;

}


let f=Math.floor(rondaGlobal/8);

let config=configuracions[f];


if(!config){

alert("Configura la franja");

return;

}


let J=getJ();

if(!J) return;


let M=[

[Number($("t1v").value)||0, Number($("t1h").value)||0],

[Number($("t2v").value)||0, Number($("t2h").value)||0]

];


// validar una sola entrada per taula

for(let i=0;i<2;i++){

if((M[i][0]&&M[i][1])||(!M[i][0]&&!M[i][1])){

alert("Error taula "+(i+1));

return;

}

}


// extreure

let vals=[],pos=[];

for(let i=0;i<2;i++){

for(let j=0;j<2;j++){

if(M[i][j]!==0){

vals.push(M[i][j]);

pos.push([i,j]);

}

}

}


let v1=vals[0], v2=vals[1];

let [i1,j1]=pos[0];

let [i2,j2]=pos[1];


let p1=(i1===0?(j1===0?config.t1V:config.t1H):(j1===0?config.t2V:config.t2H));

let p2=(i2===0?(j2===0?config.t1V:config.t1H):(j2===0?config.t2V:config.t2H));


// detectar compleix

function compleix(i,j){

let declarant = J[i][j]===1;

return declarant; // si està a la seva columna → compleix (model simplificat coherent amb J)

}


let c1=compleix(i1,j1);

let c2=compleix(i2,j2);


let punts=[0,0,0,0];


// EMPAT

if(v1===v2){

punts=[1,1,1,1];


}else{


// SEMPRE COMPARACIÓ (opció A)

let guanya=v1>v2?p1:p2;

let perd=v1>v2?p2:p1;


punts[guanya]=2;

punts[perd]=0;

punts[rival(guanya,config)]=0;

punts[rival(perd,config)]=2;

}


for(let i=0;i<4;i++) ranking[i]+=punts[i];


$("resultat").innerHTML=

"P1: "+punts[0]+"<br>"+

"P2: "+punts[1]+"<br>"+

"P3: "+punts[2]+"<br>"+

"P4: "+punts[3];


rondaCalculada=true;

rondaGlobal++;

}


function nova(){

$("t1v").value="";

$("t1h").value="";

$("t2v").value="";

$("t2h").value="";

rondaCalculada=false;

}
